> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Introduce `TapeView` as the formal abstraction between tape storage and context consumption, replacing the ad-hoc `windowed_entries()` + conditional `entries` parameter pattern in ContextBuilder and Pipeline.

**Architecture:** `TapeView` is a simple dataclass that holds a list of entries, source tape ID, window offset, and anchor IDs. It is created in `Pipeline._stage_build_context` via `TapeView.from_tape(tape)` and passed to `ContextBuilder.build()`. The builder signature gains `Tape | TapeView` union support while retaining full backward compatibility. No plugins need modification — View is internal to the pipeline→builder path.

**Tech Stack:** Python 3.11+, dataclasses, pytest

**Depends on:** Anchor Type System plan (Task 5+ — `__init__.py` exports Anchor). Not strictly required but recommended to implement Anchor first so that `TapeView.from_tape` can identify anchor entries via `isinstance`.

```python
# src/agentkit/tape/view.py
"""TapeView — task-oriented context window over a Tape."""

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any, Iterator

from agentkit.tape.models import Entry
from agentkit.tape.tape import Tape


@dataclass
class TapeView:
    """Assembled context window derived from a Tape.

    Tape is the full append-only history.
    TapeView is "what the consumer should see right now".
    """

    entries: list[Entry]
    source_tape_id: str
    window_start: int = 0
    anchor_ids: tuple[str, ...] = ()

    @classmethod
    def from_tape(cls, tape: Tape) -> TapeView:
        """Create a view from tape's current window position."""
        if tape.window_start > 0:
            entries = tape.windowed_entries()
        else:
            entries = list(tape)
        anchor_ids = tuple(e.id for e in entries if e.kind == "anchor")
        return cls(
            entries=entries,
            source_tape_id=tape.tape_id,
            window_start=tape.window_start,
            anchor_ids=anchor_ids,
        )

    @classmethod
    def full(cls, tape: Tape) -> TapeView:
        """Create a view containing all entries, ignoring window position."""
        return cls(
            entries=list(tape),
            source_tape_id=tape.tape_id,
            window_start=0,
        )

    def __len__(self) -> int:
        return len(self.entries)

    def __iter__(self) -> Iterator[Entry]:
        return iter(self.entries)
```

```python
    async def _stage_build_context(self, ctx: PipelineContext) -> None:
        from agentkit.tape.view import TapeView

        # Step 1: Windowing — may handoff the tape
        window_result = self._runtime.call_first(
            "resolve_context_window", tape=ctx.tape
        )
        if window_result is not None:
            if not (
                isinstance(window_result, tuple)
                and len(window_result) == 2
                and isinstance(window_result[0], int)
            ):
                logger.warning(
                    "resolve_context_window returned unexpected shape (%s), skipping windowing",
                    type(window_result).__name__,
                )
            else:
                window_start, summary_anchor = window_result
                if summary_anchor is not None and not ctx._handoff_done:
                    abs_window_start = ctx.tape.window_start + window_start
                    ctx.tape.handoff(summary_anchor, window_start=abs_window_start)
                    ctx._handoff_done = True
                logger.info(
                    "Context window advanced: %d entries visible (of %d total)",
                    len(ctx.tape.windowed_entries()),
                    len(ctx.tape),
                )
        else:
            # Fallback to legacy summarize_context for backward compatibility
            summary = self._runtime.call_first("summarize_context", tape=ctx.tape)
            if summary is not None:
                ctx.tape = Tape(
                    entries=list(summary),
                    tape_id=ctx.tape.tape_id,
                    parent_id=ctx.tape.parent_id,
                )
                logger.info(
                    "Context summarized (legacy): %d entries remaining", len(ctx.tape)
                )

        # Step 2: Create View from (possibly windowed) tape
        view = TapeView.from_tape(ctx.tape)

        # Step 3: Grounding from plugins
        grounding_results = self._runtime.call_many("build_context", tape=ctx.tape)
        grounding: list[dict[str, Any]] = []
        for result in grounding_results:
            if isinstance(result, list):
                grounding.extend(result)

        # Step 4: Build LLM messages from View
        from agentkit.context.builder import ContextBuilder

        system_prompt = ctx.config.get("system_prompt", "You are a helpful assistant.")
        builder = ContextBuilder(system_prompt=system_prompt)
        ctx.messages = builder.build(view, grounding=grounding or None)
```
